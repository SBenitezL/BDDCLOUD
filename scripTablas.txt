
CREATE TABLE tblPAIS
(
	IDPAIS INT NOT NULL,
	NOMBREPAIS VARCHAR(30) NOT NULL,
	primary key(IDPAIS),
);

CREATE TABLE tblCIUDAD
(
	IDCIUDAD int NOT NULL,
	NOMBRECIUDAD varchar(40) NOT NULL,
	IDPAIS INT NOT NULL,
	primary key(IDCIUDAD),
	foreign key (IDPAIS) references tblPAIS(IDPAIS)
);

CREATE TABLE tblDESARROLLADORA
(
	IDDESARROLLADORA INT NOT NULL,
	NOMBREDESARROLLADORA VARCHAR(50) NOT NULL,
	FUNDACIONDESARROLLADORA DATE NOT NULL,
	FUNDADORESDESARROLLADORA VARCHAR(200) NOT NULL,
	NOEMPLEADOSDESARROLLADORA INT NOT NULL,
	IDCIUDAD INT NOT NULL,
	primary KEY(IDDESARROLLADORA),
	foreign KEY (IDCIUDAD) references tblCIUDAD(IDCIUDAD)
);

CREATE TABLE tblPLATAFORMA
(
	IDPLATAFORMA int NOT NULL,
	NOMBREPLATAFORMA VARCHAR(35) NOT NULL,
	primary KEY (IDPLATAFORMA)
);
CREATE TABLE tblGENERO
(
	IDGENERO INT NOT NULL,
	NOMBREGENERO VARCHAR(50) NOT NULL,
	DESCRIPCIONGENERO VARCHAR(200) NOT NULL,
	primary KEY (IDGENERO)
);
CREATE TABLE tblVIDEOJUEGOS
(
	IDVIDEOJUEGOS INT NOT NULL,
	NOMBREVIDEOJUEGOS VARCHAR(80) NOT NULL,
	LANZAMIENTOVIDEOJUEGO DATE NOT NULL,
	DURACIONVIDEOJUEGOHRS NUMERIC(3,2) NOT NULL,
	DESCRIPCIONVIDEOJUEGO VARCHAR(250) NOT NULL,
	IDDESARROLLADORA INT NOT NULL,
	primary KEY (IDVIDEOJUEGOS),
	foreign KEY (IDDESARROLLADORA) references tblDESARROLLADORA(IDDESARROLLADORA)
);
CREATE TABLE tblGENEROVIDEOJUEGOS
(	
	IDGENERO INT NOT NULL,
	IDVIDEOJUEGOS INT NOT NULL,
	primary KEY (IDGENERO,IDVIDEOJUEGOS),
	foreign KEY (IDGENERO) references tblGENERO(IDGENERO),
	foreign KEY (IDVIDEOJUEGOS) references tblVIDEOJUEGOS
);
CREATE TABLE tblPLATAFORMAVIDEOJUEGOS
(
	IDVIDEOJUEGOS INT NOT NULL,
	IDPLATAFORMA INT NOT NULL,
	primary KEY (IDVIDEOJUEGOS, IDPLATAFORMA),
	foreign KEY (IDVIDEOJUEGOS) references tblVIDEOJUEGOS(IDVIDEOJUEGOS),
	foreign KEY (IDPLATAFORMA) references tblPLATAFORMA(IDPLATAFORMA)
);
CREATE TABLE tblJUGADORESPROFESIONALES(
	IDJUGADOR INT NOT NULL,
	NICKNAMEJUGADOR VARCHAR(20) NOT NULL,
	NOMBREJUGADOR VARCHAR(50) NOT NULL,
	APELLIDOJUGADOR VARCHAR(50) NOT NULL,
	IDCIUDAD INT NOT NULL,
	FECHANACIMIENTO DATE NOT NULL,
	primary key (IDJUGADOR),
	foreign KEY (IDCIUDAD) references tblCIUDAD(IDCIUDAD)
	
);
CREATE TABLE tblVIDEOJUEGOSJUGADOS(
	IDJUGADOR INT NOT NULL,
	IDVIDEOJUEGO INT NOT NULL,
	primary key(IDJUGADOR,IDVIDEOJUEGO)
	foreign KEY (IDJUGADOR) REFERENCES tblJUGADORESPROFESIONALES(IDJUGADOR)
	foreign KEY (IDVIDEOJUEGO) REFERENCES tblVIDEOJUEGOS(IDVIDEOJUEGOS)
);
