--Procedimiento que consulta un videojuego en especifico
create procedure consultarVideoJuego(@v_id int) --Pasamos los parametros con @
as
begin
	select *
	from tblVIDEOJUEGOS
	where IDVIDEOJUEGOS=@v_id;
end;

go --Comenzamos un nuevo bloque de comandos
